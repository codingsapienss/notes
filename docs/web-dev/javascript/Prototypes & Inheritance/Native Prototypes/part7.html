Part 3F — Native Prototype Exercises

Section 1 — Property Lookup & Prototype Chain​

Exercise 1 — Basic Property Lookup​

const animal = {
  eats: true,
};

const rabbit = {
  __proto__: animal,
};

console.log(rabbit.eats);

Exercise 2 — Own Property vs Prototype​

const animal = {
  eats: true,
};

const rabbit = {
  eats: false,
  __proto__: animal,
};

console.log(rabbit.eats);

Exercise 3 — Writing Property​

const animal = {
  eats: true,
};

const rabbit = {
  __proto__: animal,
};

rabbit.eats = false;

console.log(rabbit.eats);
console.log(animal.eats);

Exercise 4 — Delete Own Property​

const animal = {
  eats: true,
};

const rabbit = {
  eats: false,
  __proto__: animal,
};

delete rabbit.eats;

console.log(rabbit.eats);

Exercise 5 — Delete Prototype Property​

const animal = {
  eats: true,
};

const rabbit = {
  __proto__: animal,
};

delete animal.eats;

console.log(rabbit.eats);

Exercise 6 — Deep Prototype Chain​

const grandParent = {
  a: 1,
};

const parent = {
  __proto__: grandParent,
};

const child = {
  __proto__: parent,
};

console.log(child.a);

Exercise 7 — Property Shadowing​

const grandParent = {
  name: "Grand",
};

const parent = {
  name: "Parent",
  __proto__: grandParent,
};

const child = {
  __proto__: parent,
};

console.log(child.name);

Exercise 8 — in Operator​

const animal = {
  eats: true,
};

const rabbit = {
  __proto__: animal,
};

console.log("eats" in rabbit);
console.log("jump" in rabbit);

Exercise 9 — hasOwnProperty​

const animal = {
  eats: true,
};

const rabbit = {
  __proto__: animal,
};

console.log(rabbit.hasOwnProperty("eats"));
console.log("eats" in rabbit);

Exercise 10 — Reading Undefined Property​

const animal = {};

const rabbit = {
  __proto__: animal,
};

console.log(rabbit.walk);

Section 2 — Constructor Functions & F.prototype​

Exercise 11 — Basic Constructor​

function Rabbit() {}

Rabbit.prototype = {
  eats: true,
};

const rabbit = new Rabbit();

console.log(rabbit.eats);

Exercise 12 — Change Prototype After Object Creation​

function Rabbit() {}

Rabbit.prototype = {
  eats: true,
};

const rabbit = new Rabbit();

Rabbit.prototype = {
  jumps: true,
};

console.log(rabbit.eats);
console.log(rabbit.jumps);

Exercise 13 — Modify Existing Prototype​

function Rabbit() {}

Rabbit.prototype = {
  eats: true,
};

const rabbit = new Rabbit();

Rabbit.prototype.eats = false;

console.log(rabbit.eats);

Exercise 14 — Delete Prototype Property​

function Rabbit() {}

Rabbit.prototype = {
  eats: true,
};

const rabbit = new Rabbit();

delete Rabbit.prototype.eats;

console.log(rabbit.eats);

Exercise 15 — Constructor Property​

function Rabbit() {}

const rabbit = new Rabbit();

console.log(rabbit.constructor === Rabbit);

Exercise 16 — Replacing Prototype​

function Rabbit() {}

Rabbit.prototype = {
  jumps: true,
};

const rabbit = new Rabbit();

console.log(rabbit.constructor === Rabbit);

Exercise 17 — Fix Constructor​

function Rabbit() {}

Rabbit.prototype = {
  jumps: true,
  constructor: Rabbit,
};

const rabbit = new Rabbit();

console.log(rabbit.constructor === Rabbit);

Exercise 18 — Create Object Using Constructor​

function Rabbit(name) {
  this.name = name;
}

const rabbit = new Rabbit("White");

const rabbit2 = new rabbit.constructor("Black");

console.log(rabbit2.name);

Section 5 — Borrowing Methods​

Exercise 31​

  0: "Hello",
  1: "World",
  length: 2,
};

obj.join = Array.prototype.join;

console.log(obj.join(" "));

Exercise 32​

const obj = {
  0: "A",
  1: "B",
  length: 2,
};

console.log(Array.prototype.join.call(obj, "-"));

Exercise 33​

const obj = {
  0: "A",
  1: "B",
  length: 2,
};

const arr = Array.prototype.slice.call(obj);

console.log(arr);

Exercise 34​

  0: 10,
  1: 20,
  length: 2,
